file/.style={attached HEAD, text ragged, minimum height=0, text width=2\g, font=\small},

}

  \node[file] (#2) [below=1\g of #1] {#3};
}

  \node[commit] (#2) at (#3) {#1};
  \pgfkeys{/child count/#2/.initial=0}
}
